\begin{frame}

\frametitle{\centerline{Admissibility}}

\vspace{\fill}

\begin{center}

A function is \textbf{admissible} if \\

\vspace{\fill}

it can be derived from other functions

\pause

$\equiv$

it does not contradict the laws of the algebra

\vspace{\fill}

\end{center}

\vspace{\fill}

\end{frame}


\begin{frame}

\frametitle{XOR is admissible}

\begin{flushright}
--- Proof by reading the truth table.
\end{flushright}

\vspace{\fill}

\begin{center}

\begin{tabular}{cc|>{\centering\arraybackslash}p{0.4in}}
$A$ & $B$ & $A \oplus B$ \\\hline
\bz{} & \bz{} & \bz{} \\
\bz{} & \bo{} & \bo{} \\
\bo{} & \bz{} & \bo{} \\
\bo{} & \bo{} & \bz{}
\end{tabular}

\vspace{\fill}

$A \oplus B \equiv \p{\overline{A} \cdot B} + \p{A \cdot \overline{B}}$

\vspace{\fill}

\end{center}

\end{frame}

\frametitle{NAND is admissible}

\vspace{\fill}

\begin{center}

\begin{tabular}{cc|>{\centering\arraybackslash}p{0.4in}}

\vspace{\fill}

\end{center}

\end{frame}
