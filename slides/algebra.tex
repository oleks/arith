\begin{frame}

\vspace{\fill}

\begin{center}

An \textbf{algebra} is a collection of functions adhering to \textbf{laws}.

\end{center}

\vspace{\fill}

\end{frame}


\begin{frame}

\vspace{\fill}

\begin{center}

The functions \textbf{NOT}, \textbf{AND}, and \textbf{OR} form a
\textbf{boolean algebra}:

\begin{align*}
\text{\textbf{NOT}} &: \mathbb{B}^1 \rightarrow \mathbb{B} \\
\text{\textbf{AND}} &: \mathbb{B}^2 \rightarrow \mathbb{B} \\
\text{\textbf{OR}}  &: \mathbb{B}^2 \rightarrow \mathbb{B}
\end{align*}

\end{center}

\vspace{\fill}

\end{frame}


\begin{frame}

\frametitle{\centerline{Boolean Algebra Laws - AND form}}

\begin{center}
\begin{tabular}{l|>{\centering\arraybackslash}p{2in}}
\textbf{Name}   & \textbf{AND Form}               \\\hline
Identity Law    & $1 \cdot A = A$                 \\
Null Law        & $0 \cdot A = 0$                 \\
Idempotent Law  & $A \cdot A = A$                 \\
Inverse Law     & $A \cdot \overline{A} = 0$      \\
Commutative Law & $A \cdot B = B \cdot A$         \\
Associative Law & %
  $\p{A \cdot B} \cdot C = A \cdot \p{B \cdot C}$ \\
Distributive Law & %
  $A + B \cdot C = \p{A + B} \cdot \p{A \cdot C}$ \\
Absorption Law & %
  $A \cdot \p{A + B} = A$ \\
De Morgan's Law & %
  $\overline{A \cdot B} = \overline{A} + \overline{B}$ \\
\end{tabular}
\end{center}

\end{frame}


\begin{frame}

\frametitle{\centerline{Boolean Algebra Laws - OR form}}

\begin{center}
\begin{tabular}{l|>{\centering\arraybackslash}p{2in}}
\textbf{Name}   & \textbf{OR Form} \\\hline
Identity Law    & $0 + A = A$ \\
Null Law        & $1 + A = 1$ \\
Idempotent Law  & $A + A = A$ \\
Inverse Law     & $A + \overline{A} = 1$ \\
Commutative Law & $A + B = B + A$ \\
Associative Law & %
  $\p{A + B} + C = A + \p{B + C}$ \\
Distributive Law & %
  $A \cdot \p{B + C} = A \cdot B + A \cdot C$ \\
Absorption Law & %
  $A + A \cdot B = A$ \\
De Morgan's Law & %
  $\overline{A + B} = \overline{A} \cdot \overline{B}$ \\
\end{tabular}
\end{center}

\end{frame}
